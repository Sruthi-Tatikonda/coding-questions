1. Print Hello for the given string "AHCECLWLXO"
A. public static void main(String[] args)
   {
	String str = "AHCECLWLXO";
	for(int i=0;i<str.length();i++)
	{
		if(i%2!=0)
			System.out.print(str.charAt(i));
	}
   }
